        var intervalObj = setInterval(changePic, 1500);

        var iArray = document.getElementsByTagName("i");

        var i = 1;



        function changePic() {

            defaultColor();

            i = i % 2 + 1;

            var j = i - 1;

            iArray[j].style.backgroundColor = "#e1251b";

            document.getElementById("advImg").src = "../../images/0806/" + i + ".jpg";

        }



        function defaultColor() {

            for (var i = 0; i < iArray.length; i++) {

                iArray[i].style.backgroundColor = "#999";

            }

        }



        for (var j = 0; j < iArray.length; j++) {

            iArray[j].onclick = function () {

                document.getElementById("advImg").src = "../../images/0806/" + this.id + ".jpg";

                var j = parseInt(this.id) - 1;

                defaultColor();

                iArray[j].style.backgroundColor = "#e1251b";

                clearInterval(intervalObj);

            }

        }



        function stop() {

            clearInterval(intervalObj);

        }

    </script>

</body>



</html>

        function demo(){
            var nowDate=new Date();
            var endDate=new Date('2020/8/06 21:00:00');
            var diffTime=parseInt((endDate.getTime()-nowDate.getTime())/1000);
            var hour,minute,second;

            second=diffTime%60;//秒
            minute=parseInt(diffTime/60)%60;
            hour=parseInt(diffTime/60/60);

            var time='剩余：'+hour+'：'+minute+'：'+second+'';
            document.body.innerHTML=time;
            setTimeout(demo,1000);
        }
    </script>
